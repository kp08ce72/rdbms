1. Create a view called "employee_info" that shows the employee's first name, last name, job title, department name, and salary

CREATE VIEW EMPLOYEE_INFO AS 
SELECT FIRST_NAME,LAST_NAME,JOB_TITLE DEPARTMENT_NAME,SALARY FROM 
HR.EMPLOYEES E JOIN HR.DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
JOIN 
HR.JOBS J ON E.JOB_ID= J.JOB_ID;

2. Add a new column called "hire_date" to the "employee_info" view

CREATE OR REPLACE VIEW EMPLOYEE_INFO AS 
SELECT FIRST_NAME,LAST_NAME,JOB_TITLE DEPARTMENT_NAME,SALARY,HIRE_DATE FROM 
HR.EMPLOYEES E JOIN HR.DEPARTMENTS D ON E.DEPARTMENT_ID=D.DEPARTMENT_ID
JOIN 
HR.JOBS J ON E.JOB_ID= J.JOB_ID;


3. Update the "employee_info" view to only show employees who have a salary greater than $50,000
